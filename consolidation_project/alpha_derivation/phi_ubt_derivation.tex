\section{The Bottleneck: What is $\Phi_{\mathrm{UBT}}$ and Why is it Missing?}
\label{sec:phi_ubt_bottleneck}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Non-circularity constraint throughout this document:}
$\alpha = 1/137$ and $n^* = 137$ are \emph{never} used as inputs.
If $R_\psi$ enters a formula, it must be fixed algebraically; if it cannot,
the approach is classified as \textbf{[DEAD END]}.
All inputs are: the UBT action $\mathcal{S}[\Theta]$,
the $\mathbb{C}\otimes\mathbb{H}$ algebra ($N_{\mathrm{eff}}=12$ proved),
$\theta_H^* = 2\pi$ (Lemma~1, proved), and standard 5D heat-kernel normalization.

\subsection{The self-consistency equation (v1.1 baseline)}

The v1.1 equation for the fine-structure constant reads
\begin{equation}
  \label{eq:alpha_sc}
  \alpha \;=\;
  \frac{\theta_H^{*2}}{4\pi\,(\mathcal{I}_\psi^*)^2}
  \Bigg[
    \Phi_{\mathrm{UBT}}\sum_B C_5^{(B)}\,\mathrm{Tr}_{\mathcal{F}_B}(Q^2)
    \;+\;
    \sum_B \frac{b_B}{4\pi}
      \int_0^{\Lambda}\!\rho_B(m)\,
      \ln\!\frac{\Lambda}{m}\,dm
  \Bigg],
\end{equation}
where $\Lambda = \Phi_{\mathrm{UBT}}/Z^*$ is the UV scale.
The following quantities are \textbf{already established}:

\subsection{Numerical constraint on $\Phi_{\mathrm{UBT}}$}

With $\theta_H^{*2}/(4\pi(\mathcal{I}_\psi^*)^2) = 1/(4\pi)$
and KK spectral density $\rho(m) = 2R_\psi$ (leading order on $S^1$),
the spectral integral evaluates to
\begin{equation}
  \frac{N_{\mathrm{eff}}}{4\pi}\int_0^\Lambda 2R_\psi\,\ln\frac{\Lambda}{m}\,dm
  \;=\;
  \frac{N_{\mathrm{eff}}}{4\pi}\cdot 2R_\psi\Lambda
  \;=\;
  \frac{N_{\mathrm{eff}}}{2\pi}\;(\text{since } R_\psi\Lambda = 1).
  \label{eq:spectral_leading}
\end{equation}
Equation~\eqref{eq:alpha_sc} then requires
\begin{equation}
  4\pi\alpha
  \;=\;
  \Phi_{\mathrm{UBT}}\cdot\frac{3}{4\pi^2}
  + \frac{N_{\mathrm{eff}}}{2\pi}
  \qquad\Longrightarrow\qquad
  \Phi_{\mathrm{UBT}}
  = \frac{4\pi^2}{3}\!\left(4\pi\alpha - \frac{N_{\mathrm{eff}}}{2\pi}\right).
  \label{eq:phi_constraint}
\end{equation}
For the canonical choice $Z^* = 1/(2\pi R_\psi)$ (Hosotani normalization)
the self-consistency equation yields $\alpha \approx 1/445$
(\textbf{too small}).
For the empirically favourable $R_\psi \approx 0.917$ the equation yields
$\alpha \approx 1/137$, but this $R_\psi$ is not yet fixed algebraically —
that is the \emph{bottleneck} addressed here.

\section{Approach L1: Photon Zero-Mode Norm on $M^4\times S^1$}
\label{sec:L1_zeromode}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Status: [OPEN PROBLEM — $R_\psi$ not algebraically fixed at this level.]}

\subsection{Setup}

In the Hosotani vacuum on $M^4\times S^1_\psi$ (radius $R_\psi$) the
gauge field develops a non-trivial background along the extra dimension:
\begin{equation}
  A_\psi^{(0)} = \frac{\theta_H^*}{2\pi R_\psi} = \mathrm{const}.
  \label{eq:hosotani_bg}
\end{equation}
This is the \emph{photon zero-mode} — the unique $\psi$-independent mode
of $A_\psi$ consistent with the quantised holonomy $\theta_H^* = 2\pi$.

\subsection{Zero-mode norm $Z^*$}

The zero-mode norm is the $L^2$ norm of $A_\psi^{(0)}$ over the compact
dimension, normalised by the 4D reference volume $V_4$:
\begin{equation}
  Z^*
  = \frac{1}{V_4}\int_0^{2\pi R_\psi}\!\!\bigl|A_\psi^{(0)}\bigr|^2 d\psi
  = \frac{(2\pi R_\psi)}{V_4}\left(\frac{\theta_H^*}{2\pi R_\psi}\right)^{\!2}
  = \frac{(\theta_H^*)^2}{2\pi R_\psi\, V_4}.
  \label{eq:Z_star}
\end{equation}
For the canonical UBT choice $\theta_H^* = 2\pi$ and
$V_4 = (2\pi)^4$ (dimensionless unit volume in $\Theta$-mode normalisation):
\begin{equation}
  Z^* = \frac{(2\pi)^2}{2\pi R_\psi (2\pi)^4}
      = \frac{1}{(2\pi)^3 R_\psi}.
  \label{eq:Z_star_canonical}
\end{equation}

\subsection{$\Phi_{\mathrm{UBT}}$ from $Z^*$}

With UV cutoff $\Lambda = 1/R_\psi$ (inverse circle radius) one obtains
\begin{equation}
  \Phi_{\mathrm{UBT}}
  = \Lambda Z^*
  = \frac{1}{R_\psi}\cdot\frac{1}{(2\pi)^3 R_\psi}
  = \frac{1}{(2\pi)^3 R_\psi^2}.
  \label{eq:phi_L1}
\end{equation}
\textbf{Problem:} $\Phi_{\mathrm{UBT}}$ depends on $R_\psi$, which is not
fixed by~\eqref{eq:phi_L1} itself.

\subsection{Numerical trap: $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$}

Substituting~\eqref{eq:phi_L1} into the self-consistency equation and
requiring $\alpha = 1/137$ determines $R_\psi$ numerically.
One finds
\begin{equation}
  R_\psi \approx 0.917,
\end{equation}
which is close (error $\approx 1\%$) to the algebraically suggestive
expression
\begin{equation}
  R_\psi^{\mathrm{alg}}
  = \frac{12\pi}{N_{\mathrm{eff}}^{3/2}}
  = \frac{N_{\mathrm{eff}}\,\pi}{N_{\mathrm{eff}}\,\sqrt{N_{\mathrm{eff}}}}
  = \frac{\pi}{\sqrt{N_{\mathrm{eff}}}}
  \overset{N_{\mathrm{eff}}=12}{=} \frac{\pi}{\sqrt{12}}
  \approx 0.907.
  \label{eq:R_psi_algebraic}
\end{equation}
The formula $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$ has a plausible algebraic
interpretation: the $S^1_\psi$ circumference
$2\pi R_\psi = 2\pi^2/\sqrt{N_{\mathrm{eff}}}$
is the ratio of the topological period $2\pi^2$ to the root of the
$\mathbb{C}\otimes\mathbb{H}$ dimension.
However, no first-principles derivation of~\eqref{eq:R_psi_algebraic}
from $\mathcal{S}[\Theta]$ has been found.

\begin{gap}
  The algebraic origin of $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$
  (or equivalently of the numerical $R_\psi \approx 0.917$) is unknown.
  If this formula were derivable from the UBT action, it would
  fix $\Phi_{\mathrm{UBT}}$ from equation~\eqref{eq:phi_L1} as the
  pure number
  \begin{equation}
    \Phi_{\mathrm{UBT}}
    \;=\; \frac{N_{\mathrm{eff}}}{(2\pi)^3\pi^2}
    \;\approx\; \frac{12}{(2\pi)^3\pi^2}
    \;\approx\; 0.0049,
    \label{eq:phi_L1_algebraic}
  \end{equation}
  completing the derivation.
  The $1\%$ discrepancy between $R_\psi \approx 0.917$ and
  $R_\psi^{\mathrm{alg}} \approx 0.907$ remains to be explained.
\end{gap}

\textbf{Conclusion for L1:}
The photon zero-mode approach gives $\Phi_{\mathrm{UBT}} = 1/((2\pi)^3 R_\psi^2)$
and reduces the problem to fixing $R_\psi$.
The candidate $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$ is a
\textbf{[NUMERICAL OBSERVATION]} at $\approx 1\%$ accuracy,
not yet a proof.
Status: \textbf{[OPEN PROBLEM]}.

\section{Approach L2: Atiyah--Singer Index on $\mathbb{C}\otimes\mathbb{H}$ Bundle}
\label{sec:L2_AS}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Status: [DEAD END — index vanishes for flat base; curved base (K3)
computations A--C below show all routes require free topological parameters.]}

\subsection{Setup}

The biquaternionic field $\Theta \in \mathbb{C}\otimes\mathbb{H}$ defines
a complex vector bundle $\mathcal{E}_\Theta$ of rank $8$ (real rank 16)
over $M^4 \times S^1_\psi$.
The Hosotani background~\eqref{eq:hosotani_bg} twists this bundle through
the holonomy
\begin{equation}
  \mathrm{Hol}(\mathcal{E}_\Theta)
  = \exp\!\Bigl(\,i\oint_{S^1}A_\psi\,d\psi\Bigr)
  = e^{i\theta_H^*}
  = e^{2\pi i} = 1
  \quad(\text{for } \theta_H^* = 2\pi).
\end{equation}
Thus the Hosotani holonomy is \emph{trivial} for $\theta_H^* = 2\pi n$ with
$n\in\mathbb{Z}$, which means $\mathcal{E}_\Theta$ is topologically
equivalent to the trivial bundle in the $S^1_\psi$ direction.

\subsection{Atiyah--Singer index}

For the Dirac operator $D_\Theta$ associated with the $\Theta$-bundle
over $M^4\times S^1_\psi$:
\begin{equation}
  \mathrm{ind}(D_\Theta)
  = \int_{M^4\times S^1} \mathrm{ch}(\mathcal{E}_\Theta)\wedge\hat{A}(TM).
\end{equation}
On flat $\mathbb{R}^4\times S^1$:
\begin{itemize}
  \item $\hat{A}(T(\mathbb{R}^4\times S^1)) = 1$ (flat manifold).
  \item For the trivial bundle ($\theta_H^* = 2\pi$):
    $\mathrm{ch}(\mathcal{E}_\Theta) = \mathrm{rank}(\mathcal{E}_\Theta)
    + 0 + \cdots = 8$.
  \item $\chi(\mathbb{R}^4\times S^1) = 0$ (non-compact $\mathbb{R}^4$).
\end{itemize}
Therefore $\mathrm{ind}(D_\Theta) = 0$ for the canonical flat vacuum.

\subsection{Twisted case}

For a bundle with non-trivial first Chern class $c_1(\mathcal{E})$
(arising from a fractional holonomy $a = \theta_H^*/(2\pi) \notin\mathbb{Z}$),
one obtains a non-zero index:
\begin{equation}
  \mathrm{ind}(D_\Theta^a) = \int_{M^4\times S^1} c_1^{\mathrm{eff}}(\mathcal{E})\wedge\cdots
  \qquad (a\notin\mathbb{Z}).
\end{equation}
In this case $\Phi_{\mathrm{UBT}}$ could be identified with
$\mathrm{ind}(D_\Theta^a) / (\text{normalisation})$ as a topological
invariant.
However, since Lemma~1 establishes $\theta_H^* = 2\pi n$ (integer $n$),
the physical vacuum has trivial holonomy and $\mathrm{ind} = 0$.

\begin{remark}
  A non-trivial index could arise if the compact dimension carries torsion
  in its Pontryagin class (e.g.\ via a warped geometry or orbifold
  $S^1/\mathbb{Z}_2$).  This possibility is not excluded by Lemma~1
  and deserves further investigation.
\end{remark}

\subsection{Computation A: K3\,$\times$\,S$^1$ with trivial
  $\mathbb{C}\otimes\mathbb{H}$ bundle}
\label{subsec:L2_compA}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Motivation.}
K3 is the unique compact simply-connected complex surface with
$\mathrm{SU}(2)$ holonomy, mirroring the $\mathrm{SU}(2)$ structure of
$\mathbb{C}\otimes\mathbb{H}$.  It satisfies $\chi(\mathrm{K3})=24=2N_{\mathrm{eff}}$
(with $N_{\mathrm{eff}}=12$ proved) and $\hat{A}(\mathrm{K3})\neq 0$, making
it the minimal compact curved base with a non-trivial A-S index.

\textbf{Step 1: $\hat{A}(\mathrm{K3})$ from topology.}
K3 has Euler characteristic $\chi(\mathrm{K3})=24$ and signature
$\tau(\mathrm{K3})=-16$.  By the Hirzebruch signature theorem
$p_1[\mathrm{K3}] = 3\tau(\mathrm{K3}) = -48$.
The $\hat{A}$-genus of a compact spin 4-manifold is
$\hat{A} = -p_1[M]/24$, giving
\begin{equation}
  \hat{A}(\mathrm{K3}) = -\frac{-48}{24} = 2.
  \label{eq:Ahat_K3}
\end{equation}
\emph{Consistency check via Noether formula} (K3 is Calabi--Yau: $c_1=0$,
$c_2=\chi=24$):
$\chi_{\mathrm{hol}} = (c_1^2+c_2)/12 = 24/12 = 2 = \hat{A}(\mathrm{K3})$.~$\checkmark$

\textbf{Step 2: Index on K3 with trivial bundle.}
For the trivial $\mathbb{C}\otimes\mathbb{H}$ bundle $\mathcal{E}_\Theta$
of complex rank~$8$ over K3, $\mathrm{ch}(\mathcal{E}_\Theta)=8$ (only the
rank term survives for a trivial bundle).
\begin{equation}
  \mathrm{ind}(D^+_\Theta\text{ on K3})
  = \int_{\mathrm{K3}} \hat{A}(T\mathrm{K3})\,\mathrm{ch}(\mathcal{E}_\Theta)
  = \hat{A}(\mathrm{K3})\times\mathrm{rank}(\mathcal{E}_\Theta)
  = 2\times 8 = 16.
  \label{eq:ind_K3_trivial}
\end{equation}

\textbf{Step 3: Winding $n$ on K3\,$\times$\,S$^1$.}
Lemma~1 establishes $\theta_H^* = 2\pi n$ for any $n\in\mathbb{Z}\setminus\{0\}$.
For a gauge field with $n$ units of topological winding around $S^1_\psi$,
the bundle $\mathcal{E}_\Theta$ acquires first Chern class $n$ in the
$S^1_\psi$ direction.  The A-S formula on $\mathrm{K3}\times S^1$ gives
\begin{equation}
  \mathrm{ind}(D^+_\Theta,\,n)
  = n\times\mathrm{ind}(D^+_\Theta\text{ on K3})
  = 16n.
  \label{eq:ind_K3S1_n}
\end{equation}
The first values are $16,\,32,\,48,\,64,\,80,\ldots$ for $n=1,2,3,4,5,\ldots$
For $n=4$: $\mathrm{ind}=64\in[61,71]$.

\textbf{Is $n$ algebraically fixed?}
Lemma~1 permits any $n\in\mathbb{Z}\setminus\{0\}$; the
$\mathbb{C}\otimes\mathbb{H}$ algebraic structure places no further
constraint on the winding number~$n$.  It is a free integer parameter.

\textbf{Result A:}
$n=4$ gives $\mathrm{ind}=64\in[61,71]$, but $n$ is a free parameter.
No fixed index in the target range.  Status: \textbf{[DEAD END]}.

\subsection{Computation B: Twisted $\mathbb{C}\otimes\mathbb{H}$ bundle
  (instanton number $k$)}
\label{subsec:L2_compB}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Setup.}
Since $\mathbb{C}\otimes\mathbb{H}\cong M_2(\mathbb{C})
=\mathrm{End}_\mathbb{C}(\mathbb{C}^2)$ as algebras, the natural
gauge-twisted bundle over K3 is $\mathcal{E}=\mathrm{End}(V)$, where
$V$ is a rank-2 $\mathrm{SU}(2)$ bundle with instanton number
$c_2(V)=k\in\mathbb{Z}$.

\textbf{Chern character of $\mathrm{End}(V)$.}
For $c_1(V)=0$ and $c_2(V)=k$, the Chern character formula
$\mathrm{ch}(V) = \mathrm{rank} + c_1 + (c_1^2-2c_2)/2 = 2+0+(0-2k)/2$
gives $\mathrm{ch}(V)=2-k$ in degrees $0$ and $4$ (i.e.\ $\mathrm{ch}_0=2$
and $\int_{\mathrm{K3}}\mathrm{ch}_2(V)=-k$); likewise $\mathrm{ch}(V^*)=2-k$
since $c_1(V^*)=0$ for SU(2).
The tensor-product formula for $\mathrm{End}(V)=V\otimes V^*$
(complex rank $4$) gives
\begin{equation}
  \int_{\mathrm{K3}}\mathrm{ch}_2(\mathrm{End}(V))
  = \mathrm{ch}_0(V)\!\int_{\mathrm{K3}}\!\mathrm{ch}_2(V^*)
    + \mathrm{ch}_0(V^*)\!\int_{\mathrm{K3}}\!\mathrm{ch}_2(V)
  = 2(-k)+2(-k) = -4k.
  \label{eq:ch2_EndV}
\end{equation}

\textbf{Atiyah--Singer index.}
\begin{equation}
  \mathrm{ind}(D^+_{\mathrm{End}(V)})
  = \underbrace{\mathrm{rank}(\mathrm{End}(V))}_{4}\cdot\hat{A}(\mathrm{K3})
    + \int_{\mathrm{K3}}\mathrm{ch}_2(\mathrm{End}(V))
  = 8-4k.
  \label{eq:ind_EndV}
\end{equation}
For the target range:
\begin{equation}
  \mathrm{ind}\in[61,71]
  \;\Longleftrightarrow\;
  -4k\in[53,63]
  \;\Longleftrightarrow\;
  k\in[-15.75,\,-13.25].
\end{equation}
The only integers in this interval are $k=-14$ (giving $\mathrm{ind}=64$)
and $k=-15$ (giving $\mathrm{ind}=68$).

\textbf{Is $k$ algebraically fixed?}
The algebra $\mathbb{C}\otimes\mathbb{H}\cong M_2(\mathbb{C})$ determines
the fibre structure of $\mathrm{End}(V)$ but places \emph{no constraint} on
the topological instanton number $k$ of the underlying bundle $V$.  Any
$k\in\mathbb{Z}$ is topologically consistent with the algebraic structure.

\textbf{Result B:}
$k\in\{-14,-15\}$ yields $\mathrm{ind}\in\{64,68\}\subset[61,71]$, but $k$
is a free topological parameter not selected by the $\mathbb{C}\otimes\mathbb{H}$
algebra.  Status: \textbf{[DEAD END]}.

  \label{eq:APS_result}
\end{equation}
This is a \emph{fixed} topological integer; no free parameter enters.

\textbf{Result C:}
$\mathrm{ind}_{\mathrm{APS}}=8$, fixed and outside the target range
$[61,71]$.  Status: \textbf{[DEAD END]}.

\subsection{Identification $R_\psi = l_{\mathrm{dS}}$}
\label{subsec:L2dS_identification}

UBT defines complex time $\tau = t + i\psi$ with the imaginary part
$\psi$ compactified on $S^1_\psi$ of radius $R_\psi$.
A compact Euclidean periodic direction gives a natural temperature through
the periodicity of the thermal Green's function:
\begin{equation}
  T_\psi = \frac{\hbar}{2\pi R_\psi}.
\end{equation}
De Sitter space has an intrinsic Gibbons--Hawking temperature
\begin{equation}
  T_{\mathrm{dS}} = \frac{\hbar}{2\pi l_{\mathrm{dS}}},
  \qquad
  l_{\mathrm{dS}} = \sqrt{\frac{3}{\Lambda}},
\end{equation}
associated with the cosmological event horizon, where $\Lambda$ is the
cosmological constant.
Both temperatures arise from the \emph{same mechanism}: a compact Euclidean
time direction with periodic boundary conditions.
Setting $T_\psi = T_{\mathrm{dS}}$ therefore identifies
\begin{equation}
  \boxed{R_\psi = l_{\mathrm{dS}}.}
  \label{eq:Rpsi_ldS}
\end{equation}
This is not a postulate but a \emph{consequence} of UBT describing a de Sitter
background: both radii are periodic Euclidean times of the same physical
origin.  The remaining task is to compute $\Lambda_{\mathrm{eff}}$ from the
UBT effective potential $V_{\mathrm{eff}}(\Theta)$ and substitute
$R_\psi = \sqrt{3/\Lambda_{\mathrm{eff}}}$ into the self-consistency
equation~\eqref{eq:alpha_sc}.

For related holographic aspects of the UBT--de Sitter correspondence see
Appendix~N (holographic--Verlinde--de Sitter connection).

% ─────────────────────────────────────────────────────────────────────────────
\subsection{L2-dS-1: $\Lambda_{\mathrm{eff}}$ from $V_{\mathrm{eff}}(\Theta)$}
\label{subsec:L2dS1}

The standard quantum-field-theory relation between the vacuum energy
density and the cosmological constant is (in Planck units $G_N=1$):
\begin{equation}
  \Lambda_{\mathrm{eff}} = \frac{-V_{\min}}{8\pi},
  \label{eq:Lambda_from_Vmin}
\end{equation}
which gives

\section{Approach L3: Zeta-Regularised Determinant of $\partial_\psi + A_\psi$}
\label{sec:L3_zetadet}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Status: [KEY RESULT — gives $R_\psi$-independent pure number
$Z^* = 2\lvert\sin(\pi a)\rvert$ for $a\notin\mathbb{Z}$;
canonical vacuum $a=1$ is degenerate (zero mode) and requires separate treatment.]}

\subsection{Setup}

Consider the first-order operator on $S^1_\psi$ of radius $R_\psi$:
\begin{equation}
  \mathcal{D}_a = \partial_\psi + A_\psi,
  \qquad A_\psi = \frac{\theta_H^*}{2\pi R_\psi},
  \qquad a := \frac{\theta_H^*}{2\pi}.
  \label{eq:D_a}
\end{equation}
The eigenvalues are
$\lambda_n = (n + a)/R_\psi$, $n\in\mathbb{Z}$,
and the $\zeta$-regularised functional determinant is
$\det(\mathcal{D}_a) = \exp(-\zeta'(0;\mathcal{D}_a))$
where
$\zeta(s;\mathcal{D}_a) = \sum_{n}\lvert\lambda_n\rvert^{-s}$.

\subsection{General result for non-integer $a$}
\label{subsec:L3_noninteger}

For $a\notin\mathbb{Z}$ all eigenvalues are non-zero.
Splitting the sum into $n\geq 0$ (positive eigenvalues, using $n+a > 0$ for
large $n$) and $n\leq -1$ (negative, $\lvert n+a\rvert = -(n+a)$ for
$n < -a$) and invoking the Hurwitz zeta function
$\zeta_H(s,q)=\sum_{k=0}^\infty(k+q)^{-s}$, one obtains
\begin{equation}
  \zeta(s;\mathcal{D}_a)
  = R_\psi^{-s}\bigl[\zeta_H(s,a) + \zeta_H(s,1-a)\bigr].
  \label{eq:zeta_split}
\end{equation}
Using the standard values
$\zeta_H(0,q) = \tfrac{1}{2}-q$ and
$\zeta_H'(0,q) = \ln\Gamma(q) - \tfrac{1}{2}\ln(2\pi)$,
one computes:
\begin{align}
  \zeta(0;\mathcal{D}_a)
  &= \zeta_H(0,a)+\zeta_H(0,1-a)
   = \bigl(\tfrac{1}{2}-a\bigr)+\bigl(\tfrac{1}{2}-(1-a)\bigr) = 0,
  \label{eq:zeta0}\\[4pt]
  \zeta'(0;\mathcal{D}_a)
  &= \underbrace{\ln(R_\psi)\cdot\zeta(0;\mathcal{D}_a)}_{=0}
     + \bigl[\ln\Gamma(a)-\tfrac{1}{2}\ln(2\pi)\bigr]
     + \bigl[\ln\Gamma(1-a)-\tfrac{1}{2}\ln(2\pi)\bigr] \notag\\
  &= \ln\bigl[\Gamma(a)\,\Gamma(1-a)\bigr] - \ln(2\pi).
  \label{eq:zetaprime0}
\end{align}
Applying Euler's reflection formula $\Gamma(a)\,\Gamma(1-a) = \pi/\sin(\pi a)$:
\begin{equation}
  \zeta'(0;\mathcal{D}_a)
  = \ln\!\frac{\pi}{\sin(\pi a)} - \ln(2\pi)
  = -\ln\!\bigl(2\sin(\pi a)\bigr).
  \label{eq:zetaprime_final}
\end{equation}
Therefore
\begin{equation}
  \boxed{
  \det(\mathcal{D}_a)
  = \exp\bigl(-\zeta'(0;\mathcal{D}_a)\bigr)
  = 2\lvert\sin(\pi a)\rvert,
  \qquad a\notin\mathbb{Z}.
  }
  \label{eq:det_main}
\end{equation}
\textbf{This is $R_\psi$-independent.}
The $R_\psi^{-s}$ prefactor in~\eqref{eq:zeta_split} drops out because
$\zeta(0)=0$ (equation~\eqref{eq:zeta0}).
The result is a pure number determined solely by $a = \theta_H^*/(2\pi)$.

\subsection{Identification $Z^* = \det(\mathcal{D}_a)$}

The photon zero-mode functional integral contributes a factor
$(\det \mathcal{D}_a)^{-1/2}$ (bosonic field) or
$(\det \mathcal{D}_a)^{+1}$ (fermionic ghost) to the path integral.
The normalised zero-mode amplitude that enters the coupling is
\begin{equation}
  Z^* = \det(\partial_\psi + A_\psi) = 2\lvert\sin(\pi a)\rvert.
  \label{eq:Z_star_zetadet}
\end{equation}
With $\Lambda = 1/R_\psi$:
\begin{equation}
  \Phi_{\mathrm{UBT}}
  = \Lambda Z^*
  = \frac{2\lvert\sin(\pi a)\rvert}{R_\psi}.
  \label{eq:phi_L3_Rpsi}
\end{equation}
This still depends on $R_\psi$ through $\Lambda = 1/R_\psi$.
However, since $Z^*(R_\psi)$ is $R_\psi$-independent and $\Lambda \sim R_\psi^{-1}$,
the product $\Phi_{\mathrm{UBT}}$ scales as $1/R_\psi$, not as a pure number.

\subsection{Canonical vacuum $a=1$ ($\theta_H^* = 2\pi$): zero-mode problem}

For $\theta_H^* = 2\pi$ (i.e.\ $a=1$), the eigenvalue $\lambda_{-1} = 0$
is a \emph{zero mode} and $\det(\mathcal{D}_1) = 2\sin(\pi) = 0$.
The reduced determinant $\det'(\mathcal{D}_1)$ (excluding the zero mode)
is computed separately:

\noindent With $a=1$ and $n=-1$ excluded, the eigenvalues split as
$|n+1|$ for $n\geq 0$ (giving $\zeta_R(s)$) and for $n\leq -2$ (also
giving $\zeta_R(s)$), so:
\begin{equation}
  \zeta'(0;\mathcal{D}_1')
  = 2\zeta_R'(0) - \ln(R_\psi)\cdot 2\zeta_R(0)
  = 2\!\left(-\tfrac{1}{2}\ln(2\pi)\right) - \ln(R_\psi)\cdot(-1)
  = -\ln(2\pi) + \ln(R_\psi)
  = \ln\!\frac{R_\psi}{2\pi}.
\end{equation}
Therefore
\begin{equation}
  \det'(\mathcal{D}_1) = \frac{2\pi}{R_\psi},
  \label{eq:detprime_a1}
\end{equation}
and $\Phi_{\mathrm{UBT}} = \Lambda\,\det'(\mathcal{D}_1)
= (1/R_\psi)\cdot(2\pi/R_\psi) = 2\pi/R_\psi^2$,
which again depends on $R_\psi$.

\subsection{Numerical check}

For non-integer $a$ near $1$, equation~\eqref{eq:det_main} gives
$Z^* = 2\sin(\pi a)$.
The self-consistency equation~\eqref{eq:alpha_sc}
(with $1/(4\pi)$ prefactor and spectral term) combined
with $\Lambda Z^* = \Phi_{\mathrm{UBT}}$ requires
$\Phi_{\mathrm{UBT}} \approx 61$--$71$ (from the numerical analysis).
For $Z^*$ of order unity (e.g.\ $Z^* = 2\sin(\pi a)$ for generic $a$),
one needs $\Lambda \approx 61$--$71$, which means
$R_\psi \approx 1/65 \approx 0.015$.
This is far from the phenomenologically preferred $R_\psi \approx 0.917$,
confirming that the $Z^* = \det(\mathcal{D}_a)$ identification is
\textbf{not directly consistent} with the v1.1 equation at the level of
the algebraic part alone.

\begin{remark}
  The result~\eqref{eq:det_main} is nonetheless significant:
  it provides the \emph{only known $R_\psi$-independent quantity}
  derivable from $\mathcal{S}[\Theta]$ via zeta regularisation.
  If the definition $\Phi_{\mathrm{UBT}} = Z^* := 2|\sin(\pi a)|$ is adopted
  (without the $\Lambda$ factor), then the self-consistency equation
  fixes $R_\psi$ through $\Lambda = \Phi_{\mathrm{UBT}}/Z^* = 1$.
  This interpretation shifts the problem to explaining the unit value
  of $\Lambda$ in natural UBT units.
\end{remark}

\textbf{Conclusion for L3:}
The zeta-det approach yields the key result $\det(\mathcal{D}_a) = 2|\sin(\pi a)|$,
which is $R_\psi$-independent and topologically determined.
For the canonical vacuum $a=1$ a zero mode exists; $\det' = 2\pi/R_\psi$
still depends on $R_\psi$.
If $Z^* = 2|\sin(\pi a)|$ is the correct identification, the numerical
value $Z^* \leq 2$ is much smaller than the required $\Phi_{\mathrm{UBT}}\approx
61$--$71$ unless a large $\Lambda$ compensates.
Status: \textbf{[PARTIAL — key topological formula derived,
but numerical consistency with v1.1 equation not achieved without $R_\psi$ input]}.

\section{Approach L4: Renormalization Finiteness Condition}
\label{sec:L4_renorm}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Status: [DEAD END — finiteness condition introduces renormalization

\subsection{Setup}

The photon self-energy (vacuum polarization) in 5D UBT at one loop is

  = \frac{N_{\mathrm{eff}}}{12\pi^2}
    \Bigl[\Lambda^2 - q^2\ln\!\frac{\Lambda^2}{q^2} + \mathcal{O}(q^2)\Bigr].
  \label{eq:Pi_oneloop}
\end{equation}
The photon wavefunction renormalisation at scale $\mu$ is
\begin{equation}
  Z_3(\mu)
  = 1 - \frac{N_{\mathrm{eff}}}{12\pi^2}\ln\!\frac{\Lambda}{\mu}.
  \label{eq:Z3}
\end{equation}
The zero-mode norm from the renormalized propagator residue is
\begin{equation}
  Z^*(\mu)
  = \lim_{q^2\to 0}\frac{\Pi(q^2)}{q^2}
  = \frac{N_{\mathrm{eff}}}{12\pi^2}\ln\!\frac{\Lambda}{\mu}.
  \label{eq:Z_star_renorm}
\end{equation}

\subsection{Finiteness condition}

For $\Phi_{\mathrm{UBT}} = \Lambda Z^*(\mu)$ to be UV-finite
(independent of $\Lambda$ as $\Lambda\to\infty$), one requires
\begin{equation}
  Z^*(\mu) \;\sim\; \frac{c}{\Lambda}
  \quad\text{for some constant } c.
\end{equation}
From~\eqref{eq:Z_star_renorm} this demands
$\ln(\Lambda/\mu) \sim 1/\Lambda$, i.e.\
$\Lambda/\mu = e^{c'\!/\Lambda}\approx 1 + c'/\Lambda$,
which means $\mu \approx \Lambda$.
In other words, finiteness of $\Phi_{\mathrm{UBT}}$ requires the
renormalization scale $\mu$ to track the cutoff.

\subsection{Renormalization group consistency}

Setting $\mu = \mu_*$ such that $\Phi_{\mathrm{UBT}}(\mu_*) = \mathrm{const}$
gives the condition
\begin{equation}
  \frac{d}{d\mu}\Bigl[\Lambda\,Z^*(\mu)\Bigr] = 0
  \;\;\Longrightarrow\;\;
  \frac{d Z^*}{d\ln\mu} = 0
  \;\;\Longrightarrow\;\;
  -\frac{N_{\mathrm{eff}}}{12\pi^2} = 0.
  \label{eq:finiteness_impossible}
\end{equation}
This is impossible for $N_{\mathrm{eff}}\neq 0$.
The only escape is a cancellation between bosonic and fermionic
contributions such that the net $b_{\mathrm{eff}} = 0$,
which requires a supersymmetric spectrum not supported by the
$\mathbb{C}\otimes\mathbb{H}$ algebra with $N_{\mathrm{eff}} = 12$.

\textbf{Conclusion for L4:}
The renormalization finiteness condition does not uniquely fix $\mu$
or $\Phi_{\mathrm{UBT}}$; it merely relates $\mu$ to $\Lambda$ in a way
that reintroduces a free parameter.
Status: \textbf{[DEAD END]}.

% ─────────────────────────────────────────────────────────────────────────────

\textbf{Context.}
All four sub-tasks below probe whether the numerical candidate
$R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$ (equation~\eqref{eq:R_psi_algebraic})
follows from first principles.
Non-circularity constraint: $\alpha = 1/137$ may not enter as an input.

\subsection{1(a): Action normalization on $S^1_\psi$}

The UBT action evaluated on the fundamental winding mode
$\Theta = e^{i\psi/R_\psi}$ (winding number 1) on $S^1_\psi$ is
\begin{equation}
  \mathcal{S}[\Theta]\big|_{S^1}
  = V_4 \int_0^{2\pi R_\psi}
    \mathrm{Sc}\!\bigl[(\partial_\psi\Theta)^\dagger(\partial_\psi\Theta)\bigr]
    \,d\psi
  = V_4\cdot\frac{1}{R_\psi^2}\cdot 2\pi R_\psi
  = V_4\cdot\frac{2\pi}{R_\psi},
  \label{eq:action_winding}
\end{equation}
where $V_4$ is the 4D reference volume and
$\mathrm{Sc}[(-i/R_\psi)(i/R_\psi)] = 1/R_\psi^2$.
The ``natural scale'' condition $\mathcal{S} = 1$ requires
$R_\psi = 2\pi V_4$.
For the L1 normalisation $V_4 = (2\pi)^4$ this gives
$R_\psi = 2^5\pi^5 \approx 9740$;
for the unit volume $V_4 = 1$ one gets $R_\psi = 2\pi \approx 6.28$.
Neither matches $\pi/\sqrt{N_{\mathrm{eff}}} \approx 0.907$,
and no value of $V_4$ derivable solely from $\mathbb{C}\otimes\mathbb{H}$
algebra reproduces the algebraic candidate.

\medskip
\textbf{Status: \textbf{[DEAD END]} --- condition~\eqref{eq:action_winding}
does not fix $R_\psi$ without an independent specification of $V_4$.}

\subsection{1(c): Self-consistency from $\det'(\mathcal{D}_1)$}
\label{subsec:Rpsi_selfconsistency}

With $\Phi_{\mathrm{UBT}} = 2\pi/R_\psi^2$
(from $\Lambda = 1/R_\psi$ and $Z^* = \det'(\mathcal{D}_1) = 2\pi/R_\psi$,
equation~\eqref{eq:detprime_a1}), and KK spectral density $\rho(m) = 2R_\psi$,
the spectral integral with $b_B = N_{\mathrm{eff}}/(6\pi)$ evaluates as
\begin{equation}
  \mathcal{I}_{\mathrm{sp}}
  := \frac{b_B}{4\pi}
     \int_0^{1/R_\psi} 2R_\psi\,\ln\!\frac{1}{R_\psi m}\,dm
  = \frac{b_B}{4\pi}\cdot 2R_\psi\cdot\frac{1}{R_\psi}
  = \frac{b_B}{2\pi}
  = \frac{N_{\mathrm{eff}}}{12\pi^2}
  = \frac{1}{\pi^2},
  \label{eq:spectral_exact}
\end{equation}
where $\int_0^\Lambda \ln(\Lambda/m)\,dm = \Lambda$ was used
together with $R_\psi \Lambda = 1$.
The self-consistency equation~\eqref{eq:alpha_sc} then reads
\begin{equation}
  4\pi\alpha
  = \frac{2\pi}{R_\psi^2}\cdot\frac{3}{4\pi^2}
    + \frac{1}{\pi^2}
  = \frac{3}{2\pi R_\psi^2} + \frac{1}{\pi^2}.
  \label{eq:alpha_Rpsi}
\end{equation}
Treating $R_\psi$ as the unknown and imposing the UV normalisation
$\alpha = 1$ (strong coupling at the UBT cutoff scale, i.e.\ the
bracket in the self-consistency equation equals $4\pi$):
\begin{align}
  4\pi &= \frac{3}{2\pi R_\psi^2} + \frac{1}{\pi^2}
  \notag\\
  \frac{3}{2\pi R_\psi^2} &= 4\pi - \frac{1}{\pi^2} = \frac{4\pi^3-1}{\pi^2}
  \notag\\
  R_\psi^2 &= \frac{3\pi}{2(4\pi^3-1)} = \frac{3\pi}{8\pi^3-2}
  \qquad\Longrightarrow\qquad
  R_\psi^{\mathrm{SC}}
  = \sqrt{\frac{3\pi}{8\pi^3-2}}
  \approx 0.196.
  \label{eq:Rpsi_selfcons}
\end{align}
The ratio to the algebraic candidate is
$R_\psi^{\mathrm{alg}}/R_\psi^{\mathrm{SC}}
= (\pi/\sqrt{12})/0.196 \approx 4.6$;
the two values are \emph{not equal}.

Furthermore, from~\eqref{eq:alpha_Rpsi}, the physical requirement
$\alpha = 1/137$ would demand
\begin{equation}
  \frac{3}{2\pi R_\psi^2}
  = \frac{4\pi}{137} - \frac{1}{\pi^2}
  \approx 0.0917 - 0.1013
  = -0.0096 < 0,
  \label{eq:alpha137_impossible}
\end{equation}
which is impossible for real positive $R_\psi$.
The L3 reduced-determinant formula $\Phi_{\mathrm{UBT}} = 2\pi/R_\psi^2$,
combined with the spectral term~\eqref{eq:spectral_exact},
\emph{cannot} produce $\alpha = 1/137$ for any positive $R_\psi$.

\medskip
\textbf{Status: \textbf{[DEAD END]} ---
$R_\psi^{\mathrm{SC}} \approx 0.196 \neq \pi/\sqrt{N_{\mathrm{eff}}} \approx 0.907$
(discrepancy $\approx 78\%$);
moreover, $\alpha = 1/137$ is algebraically unreachable from the L3
formula with $b_B = N_{\mathrm{eff}}/(6\pi) > 0$.}

\subsection{1(d): Modular invariance of $S^1_\psi$}
\label{subsec:modular_invariance}

The compact imaginary-time circle $S^1_\psi$ of radius $R_\psi$ carries
the Euclidean modular parameter $\tau = iR_\psi$ (purely imaginary).
Under the modular $S$-transformation $\tau \to -1/\tau$:
\begin{equation}
  iR_\psi \;\xrightarrow{\;S\;}\; -\frac{1}{iR_\psi} = \frac{i}{R_\psi}.
  \label{eq:S_transform}
\end{equation}
The self-dual (S-fixed) point requires $iR_\psi = i/R_\psi$, i.e.\
$R_\psi^2 = 1$:
\begin{equation}
  R_\psi^{S\text{-dual}} = 1.
  \label{eq:Sdual}
\end{equation}
Checking whether the candidate $\tau^* = i\pi/\sqrt{N_{\mathrm{eff}}}$
satisfies any modular relation:
$-1/\tau^* = i\sqrt{N_{\mathrm{eff}}}/\pi \neq \tau^*$
unless $\pi^2 = N_{\mathrm{eff}} = 12$;
however $\pi^2 \approx 9.870 \neq 12$ (discrepancy $\approx 18\%$).
The T-duality partner of $R_\psi = \pi/\sqrt{12}$ is
$\tilde{R}_\psi = \sqrt{12}/\pi \approx 1.103$,
which is also not algebraically distinguished.

\medskip
\textbf{Status: \textbf{[DEAD END]} ---
modular invariance fixes $R_\psi = 1$ (S-self-dual point).
The candidate $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$ would require
$\pi^2 = N_{\mathrm{eff}}$, which is false by $\approx 18\%$.}

\section{Zero-Mode Regulation}
\label{sec:zeromode_regulation}
% ─────────────────────────────────────────────────────────────────────────────

\textbf{Context.}
The canonical vacuum $\theta_H^* = 2\pi$ ($a = 1$) forces a zero mode
$\lambda_{-1} = 0$ in $\mathcal{D}_1$, making $\det(\mathcal{D}_1) = 0$.
The reduced determinant $\det'(\mathcal{D}_1) = 2\pi/R_\psi$ removes the
zero mode but introduces $R_\psi$ dependence.
Three standard regulation prescriptions are analysed below.

\subsection{2(a): Collective coordinate (instanton method)}
\label{subsec:collective_coord}

The zero mode of $\mathcal{D}_1$ corresponds to the flat direction of
the Hosotani vacuum: holonomy shifts $\theta_H \to \theta_H + \delta$
cost no action.
In the instanton calculus the zero-mode integration is extracted as
the collective coordinate $\theta_H$:
\begin{equation}
  \int_0^{2\pi} d\theta_H\,e^{-S_{\mathrm{eff}}(\theta_H)}
  = \int_0^{2\pi} 1\,d\theta_H = 2\pi,
  \label{eq:cc_volume}
\end{equation}
where $S_{\mathrm{eff}} = 0$ along the flat direction.
The regulated amplitude is
\begin{equation}
  Z^*_{\mathrm{cc}}
  := \det'(\mathcal{D}_1)\times 2\pi
  = \frac{2\pi}{R_\psi}\cdot 2\pi
  = \frac{4\pi^2}{R_\psi},
  \label{eq:Zcc}
\end{equation}
giving
\begin{equation}
  \Phi_{\mathrm{UBT}}^{(\mathrm{cc})}
  = \Lambda\,Z^*_{\mathrm{cc}}
  = \frac{1}{R_\psi}\cdot\frac{4\pi^2}{R_\psi}
  = \frac{4\pi^2}{R_\psi^2}.
  \label{eq:phi_cc}
\end{equation}
For the algebraic candidate $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$:
\begin{equation}
  \Phi_{\mathrm{UBT}}^{(\mathrm{cc})}
  = \frac{4\pi^2\,N_{\mathrm{eff}}}{\pi^2}
  = 4N_{\mathrm{eff}} = 48.
  \label{eq:phi_cc_value}
\end{equation}
The result $48 = 4 \times 12$ is a pure integer multiple of
$N_{\mathrm{eff}}$ and is $\pi$-free, which is algebraically elegant.
However, it requires $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$ as input.

\medskip
\textbf{Status: \textbf{[NUMERICAL OBSERVATION]} ---
$\Phi^{(\mathrm{cc})} = 4N_{\mathrm{eff}} = 48$ for
$R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$;
$R_\psi$ is not independently derived.
Numerical $\alpha$ check: Section~\ref{sec:numerical_consistency}.}

\subsection{2(b): Limit $a \to 1^-$}
\label{subsec:lim_a1}

For $a \to 1^-$ (approaching the degenerate vacuum from the non-integer
side), $\det(\mathcal{D}_a) = 2\sin(\pi a) \to 0^+$.
The ratio to the singular factor has a finite limit:
\begin{equation}
  \lim_{a\to 1^-}
  \frac{\det(\mathcal{D}_a)}{|\sin(\pi a)|}
  = \lim_{a\to 1^-} 2 = 2.
  \label{eq:ratio_limit}
\end{equation}
Defining the renormalised zero-mode amplitude
$Z^*_{\mathrm{ren}} := 2$ ($R_\psi$-independent),
the corresponding $\Phi_{\mathrm{UBT}}$ is
\begin{equation}
  \Phi_{\mathrm{UBT}}^{(\mathrm{ren})}
  = \Lambda\,Z^*_{\mathrm{ren}}
  = \frac{2}{R_\psi}.
  \label{eq:phi_ren}
\end{equation}
For $R_\psi = \pi/\sqrt{N_{\mathrm{eff}}}$:
$\Phi_{\mathrm{UBT}}^{(\mathrm{ren})} = 2\sqrt{N_{\mathrm{eff}}}/\pi
= 2\sqrt{12}/\pi \approx 2.21$.

\medskip
\textbf{Status: \textbf{[DEAD END]} ---
the $a \to 1^-$ limit yields the universal factor $Z^* = 2$
but leaves $\Phi_{\mathrm{UBT}} = 2/R_\psi$ $R_\psi$-dependent.}

\subsection{2(c): $\mathbb{Z}_2$ orbifolding $S^1_\psi/\mathbb{Z}_2$}
\label{subsec:orbifold}

Under the $\mathbb{Z}_2$ identification $\psi \to -\psi$, the circle
$S^1_\psi$ is reduced to the interval $[0,\pi R_\psi]$.
Anti-periodic boundary conditions correspond to the twisted sector with
effective holonomy parameter
\begin{equation}
  a_{\mathrm{eff}} = a + \tfrac{1}{2}
  \big|_{a=1}
  = \tfrac{3}{2} \notin \mathbb{Z},
  \label{eq:a_eff}
\end{equation}
so the zero mode is absent and the full zeta-det formula gives
\begin{equation}
  Z^*_{\mathrm{orb}}
  = \det\!\left(\mathcal{D}_{3/2}\right)
  = 2\left|\sin\!\tfrac{3\pi}{2}\right|
  = 2\cdot 1 = 2.
  \label{eq:Zorb}
\end{equation}
With $\Lambda_{\mathrm{orb}} = 1/R_\psi$ (the UV cutoff remains
the inverse circle radius):
\begin{equation}
  \Phi_{\mathrm{UBT}}^{(\mathrm{orb})}
  = \frac{2}{R_\psi},
  \label{eq:phi_orb}
\end{equation}
identical to the renormalised-limit result~\eqref{eq:phi_ren}.
The untwisted sector ($a_{\mathrm{eff}} = 1$) still contains the zero
mode and requires collective-coordinate treatment as in
Section~\ref{subsec:collective_coord}.

\begin{remark}
  The coincidence $Z^*_{\mathrm{orb}} = Z^*_{\mathrm{ren}} = 2$ arises
  because both prescriptions regulate $\det = 2$ while keeping
  $\Lambda = 1/R_\psi$; the $R_\psi$-dependence of
  $\Phi_{\mathrm{UBT}}$ is therefore unavoidable in both cases.
\end{remark}

\medskip
\textbf{Status: \textbf{[DEAD END]} ---
orbifold twisted sector gives $\Phi^{(\mathrm{orb})} = 2/R_\psi$,
same as the $a \to 1^-$ limit; $R_\psi$-dependence not eliminated.}

  $R_\psi \approx 0.917$.
  The tension between L1 and L3 identifications of $\Phi_{\mathrm{UBT}}$
  is an open question.
\end{remark}

\medskip
\textbf{Summary: all L3-regulated candidates yield
$\alpha^{-1} \in \{3, 18, 47\}$, far below $137$.
The constraint~\eqref{eq:phi_required} constitutes a
\textbf{[STRUCTURAL OBSTRUCTION]} within the present framework.}

\subsection*{Main finding of this document}

\begin{theorem}[Zeta-det formula]
  For the operator $\mathcal{D}_a = \partial_\psi + a/R_\psi$ on
  $S^1_\psi$ with $a = \theta_H^*/(2\pi)\notin\mathbb{Z}$,
  the $\zeta$-regularised functional determinant satisfies
  \begin{equation}
    \det(\mathcal{D}_a) = 2\lvert\sin(\pi a)\rvert,
    \label{eq:thm_zeta}
  \end{equation}
  independently of $R_\psi$.
  This is a pure number determined entirely by the Hosotani phase
  $\theta_H^*$.
\end{theorem}

\begin{proof}
  See computation in Section~\ref{subsec:L3_noninteger}:
  $\zeta(0;\mathcal{D}_a)=0$ eliminates all $R_\psi$ dependence,
  and $\zeta'(0;\mathcal{D}_a) = -\ln(2|\sin(\pi a)|)$ follows from the
  Hurwitz values $\zeta_H(0,q)=\frac{1}{2}-q$ and
  $\zeta_H'(0,q)=\ln\Gamma(q)-\frac{1}{2}\ln(2\pi)$
  combined with Euler's reflection formula.
\end{proof}

\subsection*{Open problem}

  $\mathcal{S}[\Theta]$ alone.

  that forces a zero mode, preventing a direct application of the
  $\det = 2|\sin(\pi a)|$ formula.
  Sections~\ref{sec:algebraic_Rpsi}--\ref{sec:numerical_consistency}
  have now analysed all immediate sub-problems; the remaining
  open questions are:
  \begin{enumerate}
    \item \emph{Algebraic origin of $R_\psi$ --- all four routes are dead ends.}
          Sub-tasks 1(a)--(d) (action normalisation, proved-quantity combinations,
          self-consistency from $\det'$, and modular invariance) all conclude
          \textbf{[DEAD END]}.
          The self-consistency equation with $b_B = N_{\mathrm{eff}}/(6\pi)$
          gives $R_\psi^{\mathrm{SC}} = \sqrt{3\pi/(8\pi^3-2)} \approx 0.196
          \neq \pi/\sqrt{N_{\mathrm{eff}}} \approx 0.907$ (a factor $\approx 4.6$ discrepancy),
          and furthermore renders $\alpha = 1/137$ algebraically unreachable
          (equation~\eqref{eq:alpha137_impossible}).

          $\mathbb{Z}_2$ orbifold (2c) all leave $R_\psi$ as a free parameter

    \item \emph{Spectral sign: structural obstruction.}
          The bosonic KK spectral term $1/\pi^2$ exceeds $4\pi/137$,
