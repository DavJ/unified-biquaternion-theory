\section{Hermitian Slice Compatibility with QED}\label{sec:hermitian_slice}
\begin{lemma}\label{lem:hermitian_qed}
On the Hermitian slice of the biquaternion bundle, with $D_\mu \to \partial_\mu - i\frac{e}{\hbar c}A_\mu$
and $D_\tau\to \partial_t$, the emergent definition reduces to
\begin{equation}\label{eq:hermitian_alpha}
\alpha \;=\; \frac{e^2}{4\pi \hbar c}\,,
\end{equation}
with no extra factors.
\end{lemma}
\begin{proof}
In this limit $I_{\mathrm s}$ measures the U(1) vertex strength while $I_{\mathrm t}$ measures the
temporal normalization fixed by $R_\psi$. Their ratio reproduces the Thomson-limit normalization,
and by Theorem~\ref{thm:B2alpha} the extraction factor is unity.
\end{proof}
