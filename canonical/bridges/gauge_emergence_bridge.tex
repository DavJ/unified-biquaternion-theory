\subsection{\texorpdfstring{$\mathrm{SU}(3)_c$}{SU(3)c} emergence (PROVED [L0])}

\textbf{Claim:} The strong colour gauge group $\mathrm{SU}(3)_c$ arises from
the $\mathbb{Z}_2 \times \mathbb{Z}_2 \times \mathbb{Z}_2$ involution structure
of $\mathbb{C}\otimes\mathbb{H}$.

\textbf{Theorems G.A--G.D} (proved):
\begin{itemize}
  \item G.A — The Lie algebra $\mathfrak{su}(3)$ is realised in $\mathbb{C}\otimes\mathbb{H}$
  \item G.B — Quarks transform in the fundamental representation
  \item G.C — Gluons transform in the adjoint representation
  \item G.D — Electroweak and strong sectors decouple
\end{itemize}

\textbf{Location:} \texttt{canonical/interactions/sm\_gauge.tex},
Section~\ref{sec:canonical:sm_gauge}, Theorems G.A--G.D.

\textbf{Colour confinement:} Algebraically, free quarks are inadmissible and
hadrons satisfy the colour singlet condition $\langle C_2\rangle = 0$.
Status: \textbf{conjectured with experimental support [L0]} (LHCb exotic hadron
data consistent). No rigorous proof of confinement from dynamics exists (as in
standard QCD).

\subsection{Weinberg Angle \texorpdfstring{$\theta_W$}{theta\_W} (SEMI-EMPIRICAL)}

\begin{tcolorbox}[colback=orange!10!white,colframe=orange!75!black,title=Semi-empirical: $\theta_W$]

Status: \textbf{semi-empirical} — not a failure of the framework but an identified
limitation of the $\mathbb{C}\otimes\mathbb{H}$ algebraic sector.

\textbf{Location:} \texttt{canonical/interactions/sm\_gauge.tex} — explicitly
flagged as semi-empirical; \texttt{DERIVATION\_INDEX.md}: ``Weinberg angle
$\theta_W$: Semi-empirical''.
\end{tcolorbox}

\subsection{Chirality Selection: Why \texorpdfstring{$\mathrm{SU}(2)_L$}{SU(2)\_L} and not \texorpdfstring{$\mathrm{SU}(2)_R$}{SU(2)\_R} (MOTIVATED)}

\begin{tcolorbox}[colback=orange!10!white,colframe=orange!75!black,title=Motivated (not fully proved): chirality]
Chirality selection — the fact that the weak interaction is left-handed — is
motivated by $\psi$-parity in the complex-time sector $\tau = t + i\psi$.

The physical argument: under $\psi \to -\psi$ (time-reversal in imaginary sector),
left- and right-handed couplings transform differently. However, this has not been
elevated to a formal theorem in the repository.

Status: \textbf{motivated, semi-empirical} — documented as open in
\texttt{DERIVATION\_INDEX.md} and \texttt{canonical/interactions/sm\_gauge.tex}.
\end{tcolorbox}
