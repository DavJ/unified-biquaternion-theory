\section{GR Recovery Chain: $\Theta \to g \to \Gamma \to R \to \text{Einstein}$}
\label{sec:gr_chain_summary}

\begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,
  title=Theorem (GR Recovery Chain)]

The following chain of results establishes that UBT contains standard
General Relativity as a sector.

\medskip
\textbf{Step 1: $\Theta \to g_{\mu\nu}$}
\[
  g_{\mu\nu} \;=\; \frac{\Re[\partial_\mu\Theta \cdot \partial_\nu\Theta^\dagger]}{\mathcal{N}},
  \qquad \mathcal{N} > 0 \text{ (scale-fixing scalar)}.
\]
$\mathcal{N}$ determines the physical units of the metric but \textbf{does not
determine the signature} — see Step~3.
\hfill\textit{[Ref: \texttt{step1\_metric\_bridge.tex}]}

\medskip
\textbf{Step 2: $g_{\mu\nu}$ is non-degenerate}

For $\Theta \in \mathcal{A}_\mathrm{UBT}$: $\det(g) \neq 0$.
\[
  \mathcal{A}_\mathrm{UBT} \;:=\;
  \{\Theta \mid \partial_\mu\Theta \text{ linearly independent},\; \Theta \neq \text{const}\}.
\]
\hfill\textit{[Ref: \texttt{step2\_nondegeneracy.tex}, Theorem~1]}

\medskip
\textbf{Step 3: Signature $(-,+,+,+)$}

From AXIOM~B: $\tau = t + i\psi \Rightarrow \partial_\tau$ lies in the timelike
sector of $\mathrm{Cl}_{1,3}(\mathbb{R})$, giving $g_{00} < 0$, $g_{ii} > 0$.
\textbf{$\mathcal{N}$ does not fix the signature} — the Lorentzian signature is
an algebraic theorem from the axioms.
\hfill\textit{[Ref: \texttt{step3\_signature\_theorem.tex}, Theorem~2]}

\medskip
\textbf{Step 4: $g \to \Gamma \to R$ (derived GR structure, real limit)}

In the real limit, the derived metric $g_{\mu\nu} := \mathrm{Re}(\mathcal{G}_{\mu\nu})$
determines the Levi-Civita connection via metric compatibility; Riemann $\to$ Ricci $\to$
Einstein tensor via standard differential geometry.
These classical GR structures are obtained as projections from UBT, not fundamental.
\hfill\textit{[Standard differential geometry, applied to derived $g$]}

\medskip
\textbf{Step 5: Einstein field equations}
\[
  \frac{\delta S_\mathrm{total}[g,\Theta]}{\delta g^{\mu\nu}} \;=\; 0
  \;\;\Longrightarrow\;\; G_{\mu\nu} \;=\; 8\pi G\, T_{\mu\nu}.
\]
$T_{\mu\nu}$ from the Hilbert prescription $\delta S_\mathrm{matter}/\delta g^{\mu\nu} =
-(\sqrt{-g}/2)\,T_{\mu\nu}$.
\hfill\textit{[Ref: \texttt{T\_munu\_derivation/step3\_einstein\_with\_matter.tex}]}

\medskip
\textbf{Step 6 [ON-SHELL]: Fully emergent gravity}

After substituting $g = g[\Theta]$: Einstein equations become equations for
$\Theta$ alone. \\
\textit{Status:} \textbf{PROVED ON-SHELL} for $\Theta \in \mathcal{A}_\mathrm{UBT}$. \\
Off-shell: requires global injectivity of $\Theta \mapsto g[\Theta]$
\hfill\textbf{[OPEN L2]}

\hfill\textit{[Ref: \texttt{step2\_theta\_only\_closure.tex}]}

\end{tcolorbox}

\begin{theorem}[Full Schwarzschild metric from $\Theta_0$, \textbf{Proved [L1]}]
\label{thm:schwarzschild_canonical}
The ansatz $\Theta_0 = e^{i\Phi(r)}\cdot[f(r)\cdot\mathbf{1}+g(r)\cdot\boldsymbol{e}_r]$

and matching condition $\alpha(r) = \Phi(r) = (1-M/2r)/(1+M/2r)$ gives the

\[
  g_{tt} = -\Phi(r)^2,\qquad g_{ij} = \Psi(r)^4\,\delta_{ij},
  \qquad \Psi(r) = 1 + M/(2r).
\]
The phase $e^{i\alpha}$ cancels in the spatial sector via
$\mathrm{Sc}(e^{-i\alpha}Ae^{i\alpha}) = \mathrm{Sc}(A)$; the temporal
component emerges from $\partial_\psi\Theta_0 = i\alpha(r)\Theta_0$ via
the complex-time structure (Step~3).
Numerically verified: \texttt{tools/verify\_schwarzschild\_theta.py},
relative error $< 10^{-8}$.
Source: \texttt{research\_tracks/research/schwarzschild\_from\_theta.tex §5};
canonical: \texttt{canonical/geometry/biquaternionic\_vacuum\_solutions.tex §3}.
\end{theorem}
