\begin{equation}
\label{eq:canonical:psychon_scalar}
\psi(x) = \psi_0 + \delta\psi(x)
\end{equation}

where:
\begin{itemize}
    \item $\psi_0$ = background (vacuum) value of imaginary time
    \item $\delta\psi(x)$ = psychon excitation field
\end{itemize}

The vector psychons correspond to excitations of $(\chi, \xi)$:

\begin{equation}
\label{eq:canonical:psychon_vector}

The complete psychon field is:

\begin{equation}
\label{eq:canonical:psychon_full}
\Psi(x) = \psi(x) + i\chi(x) + j\xi(x) \in \mathbb{H}
\end{equation}

This is a quaternion-valued field representing the imaginary part of biquaternion time.

\subsection{Psychon Lagrangian}

The psychon field dynamics are governed by the Lagrangian:

\begin{equation}
\label{eq:canonical:psychon_lagrangian}
\mathcal{L}_{\text{psychon}} = \frac{1}{2} \partial_\mu \Psi \partial^\mu \Psi^\dagger - \frac{1}{2} m_\psi^2 \Psi \Psi^\dagger - V(\Psi) - \Psi J
\end{equation}

where:
\begin{itemize}
    \item $m_\psi$ = psychon mass scale
    \item $V(\Psi)$ = self-interaction potential
    \item $J(x)$ = consciousness source current (coupling to neural activity)
\end{itemize}

\subsubsection{Expanded Form}

In components:
\begin{align}
\mathcal{L}_{\text{psychon}} = &\frac{1}{2}(\partial_\mu \psi \partial^\mu \psi + \partial_\mu \chi \partial^\mu \chi + \partial_\mu \xi \partial^\mu \xi) \\
&- \frac{1}{2} m_\psi^2 (\psi^2 + \chi^2 + \xi^2) \\
&- \lambda (\psi^2 + \chi^2 + \xi^2)^2 \\
&- \psi J_\psi - \chi J_\chi - \xi J_\xi

where $\lambda$ is a self-coupling constant.
