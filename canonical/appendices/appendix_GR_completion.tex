\section{Admissible Sector}\label{sec:admissible}
% -----------------------------------------------------------------

\begin{definition}[Admissible sector $\mathcal{A}_{\mathrm{UBT}}$]
\label{def:admissible}
A field configuration $\Theta$ lies in the admissible class
$\mathcal{A}_{\mathrm{UBT}}$ if it satisfies conditions C1--C6:
\begin{description}
  \item[C1] \textit{Real sector dominance}: $|h_{\mu\nu}| \ll |g_{\mu\nu}|$,
    where $h_{\mu\nu} = \mathrm{Im}(\mathcal{G}_{\mu\nu})$.
  \item[C2] \textit{Non-degeneracy}: $\det(g_{\mu\nu}) \neq 0$.
  \item[C3] \textit{Lorentzian signature}: $g_{\mu\nu}$ has signature
    $(-,+,+,+)$.
  \item[C4] \textit{Smooth biquaternionic field}: $\Theta \in C^\infty(\mathcal{M})$.
  \item[C5] \textit{Decoupling of gauge sector}: gauge fields appear only
    through $\mathcal{L}_{\mathrm{gauge}}$, not directly in the metric
    variation.
  \item[C6] \textit{No torsion}: the projected connection $\Gamma$ is
    torsion-free.
\end{description}
\end{definition}

\noindent\textbf{Source}: \texttt{canonical/gr\_limit/GR\_limit\_of\_UBT.tex \S2}.

\section{Hilbert--Einstein Variation}\label{sec:hilbert}
% -----------------------------------------------------------------

\subsection{Setup}

The gravitational part of the UBT action is:
\begin{equation}\label{eq:SEH}
  S_{\mathrm{grav}}[g]
  = \frac{1}{2\kappa}\int_{\mathcal{M}} \mathrm{d}^4x\,\sqrt{-g}\,R[g],
\end{equation}
where $g_{\mu\nu} = \mathrm{Re}(\mathcal{G}_{\mu\nu}[\Theta])$ and
$R = g^{\mu\nu}R_{\mu\nu}$.

\subsection{Metric Variation}

\begin{proposition}[Variation of $S_{\mathrm{grav}}$]
\label{prop:hilbert}
  \begin{equation}\label{eq:varSEH}
    \delta S_{\mathrm{grav}}
    = \frac{1}{2\kappa}\int \mathrm{d}^4x\,\sqrt{-g}\,
      \!\left(R_{\mu\nu} - \tfrac{1}{2}g_{\mu\nu}R\right)\delta g^{\mu\nu}.
  \end{equation}
\end{proposition}

\begin{proof}
Standard result: $\delta(\sqrt{-g}\,R)/\delta g^{\mu\nu}
= \sqrt{-g}(R_{\mu\nu} - \frac12 g_{\mu\nu}R)$.
The boundary term (variation of $\Gamma$) vanishes in the bulk.
$\square$
\end{proof}

\subsection{Matter Variation}

The matter action $S_{\mathrm{mat}} = \int\sqrt{-g}\,(\mathcal{L}_{\mathrm{kin}}
+ \mathcal{L}_{\mathrm{gauge}} + \mathcal{L}_{\mathrm{pot}})$ contributes:
\begin{equation}\label{eq:deltaSmatter}
  \delta S_{\mathrm{mat}}
  = -\tfrac{1}{2}\int \mathrm{d}^4x\,\sqrt{-g}\,T_{\mu\nu}\,\delta g^{\mu\nu},
\end{equation}
where the stress-energy tensor is:
\begin{equation}\label{eq:Tdef}
  T_{\mu\nu}
  := -\frac{2}{\sqrt{-g}}\frac{\delta(\sqrt{-g}\,\mathcal{L}_{\mathrm{mat}})}
     {\delta g^{\mu\nu}}.
\end{equation}

\section{Einstein Equation Recovery}\label{sec:einstein}
% -----------------------------------------------------------------

\begin{theorem}[GR recovery on $\mathcal{A}_{\mathrm{UBT}}$]
\label{thm:GR}
On the admissible sector $\mathcal{A}_{\mathrm{UBT}}$ (Definition~\ref{def:admissible}),
the total variation $\delta(S_{\mathrm{grav}} + S_{\mathrm{mat}}) = 0$ yields:
\begin{equation}\label{eq:Einstein}
  \boxed{
  R_{\mu\nu} - \tfrac{1}{2}g_{\mu\nu}R
  \;=\; \kappa\,T_{\mu\nu}.
  }
\end{equation}
This is exactly Einstein's field equation with $\kappa = 8\pi G/c^4$.
\end{theorem}

\begin{proof}
Combining Propositions~\ref{prop:hilbert} and~\eqref{eq:deltaSmatter},
\begin{equation}
  \delta S = \int\sqrt{-g}\!\left[\frac{1}{2\kappa}\!\left(R_{\mu\nu}
  - \tfrac12 g_{\mu\nu}R\right) - \tfrac12 T_{\mu\nu}\right]\delta g^{\mu\nu}
  = 0.
\end{equation}
Since $\delta g^{\mu\nu}$ is arbitrary, the bracket vanishes, giving
Eq.~\eqref{eq:Einstein}.  The map $\Theta \mapsto g_{\mu\nu}[\Theta]$
introduces a Jacobian $J^*$ coupling the $\Theta$ and metric equations
(see \texttt{GR\_limit\_of\_UBT.tex \S3.2}); on $\mathcal{A}_{\mathrm{UBT}}$
the conditions C1--C6 ensure that this coupling does not deform the
Einstein equation. $\square$
\end{proof}

\noindent\textbf{Source}: \texttt{canonical/gr\_limit/GR\_limit\_of\_UBT.tex \S4}.

\begin{remark}
Outside $\mathcal{A}_{\mathrm{UBT}}$, the imaginary sector $h_{\mu\nu}$ may
contribute additional terms to the field equations, representing genuine UBT
corrections beyond GR.
\end{remark}

\section{Explicit Form of $T_{\mu\nu}$}\label{sec:tmunu}
% -----------------------------------------------------------------

From~\eqref{eq:Tdef} applied to $\mathcal{L}_{\mathrm{kin}}$:

\begin{equation}\label{eq:Tkin}
  T^{(\mathrm{kin})}_{\mu\nu}
  = \mathrm{Re}\,\mathrm{Tr}\!\left[
    (D_\mu\Theta)^\dagger D_\nu\Theta
    + (D_\nu\Theta)^\dagger D_\mu\Theta
    - g_{\mu\nu}(D^\rho\Theta)^\dagger D_\rho\Theta
  \right].
\end{equation}

For the gauge sector $\mathcal{L}_{\mathrm{gauge}} = -\frac14 F^a_{\mu\nu}F^{a\mu\nu}$:
\begin{equation}\label{eq:Tgauge}
  T^{(\mathrm{gauge})}_{\mu\nu}
  = F^a_{\mu\rho}F^{a\,\rho}{}_\nu
  - \tfrac{1}{4}g_{\mu\nu}F^a_{\rho\sigma}F^{a\,\rho\sigma}.
\end{equation}

The total stress-energy tensor is:
\begin{equation}\label{eq:Ttotal}
  T_{\mu\nu} = T^{(\mathrm{kin})}_{\mu\nu} + T^{(\mathrm{gauge})}_{\mu\nu}
  + T^{(\mathrm{pot})}_{\mu\nu},
\end{equation}
where $T^{(\mathrm{pot})}_{\mu\nu} = -g_{\mu\nu}V(\Theta)$ for a
field-space potential $V$.
