\section{Biquaternionic Metric Tensor}\label{sec:tensor}
% -----------------------------------------------------------------

\subsection{Fundamental Construction}

\begin{definition}[Biquaternionic metric]\label{def:biqmetric}
The biquaternionic metric tensor is constructed from the fundamental
field $\Theta(q,\tau) \in \mathbb{C}\otimes\mathbb{H}$ as:
\begin{equation}\label{eq:Gmunu}
  \mathcal{G}_{\mu\nu}[\Theta]
  \;:=\;
  \mathrm{Tr}\!\bigl(\partial_\mu\Theta \cdot \partial_\nu\Theta^\dagger\bigr)
  \;\in\; \mathbb{C}\otimes\mathbb{H}.
\end{equation}
\end{definition}

\noindent\textbf{Source}: \texttt{canonical/geometry/biquaternion\_metric.tex \S1--\S2}.

\subsection{Symmetry}

\begin{proposition}[Symmetry of $\mathcal{G}_{\mu\nu}$]\label{prop:sym}
  $\mathcal{G}_{\mu\nu} = \mathcal{G}_{\nu\mu}^*$
  (Hermitian symmetry).  In the real projection,
  $g_{\mu\nu} = g_{\nu\mu}$.
\end{proposition}

\begin{proof}
$\mathrm{Tr}(\partial_\mu\Theta\,\partial_\nu\Theta^\dagger)
 = [\mathrm{Tr}(\partial_\nu\Theta\,\partial_\mu\Theta^\dagger)]^*$
by the cyclic property of the trace and the involution
$(\partial\Theta)^\dagger\mapsto\partial\Theta^\dagger$.
$\square$
\end{proof}

\subsection{Real Projection}

\begin{definition}[Classical metric]\label{def:classicmetric}
The classical (GR) metric tensor is the real part of the biquaternionic metric:
\begin{equation}\label{eq:gmunu}
  \boxed{g_{\mu\nu} \;:=\; \mathrm{Re}\!\bigl(\mathcal{G}_{\mu\nu}[\Theta]\bigr).}
\end{equation}
\end{definition}

\begin{remark}
The imaginary part $h_{\mu\nu} := \mathrm{Im}(\mathcal{G}_{\mu\nu})$ encodes
phase curvature invisible to classical observers coupling only to the real sector.
\end{remark}

\section{Coordinate Covariance}\label{sec:covariance}
% -----------------------------------------------------------------

\begin{theorem}[Tensorial character of $g_{\mu\nu}$]\label{thm:tensor}
  Under smooth coordinate changes $x^\mu \to x'^\mu(x)$,
  $g_{\mu\nu}$ transforms as a covariant rank-2 tensor:
  \begin{equation}
    g'_{\mu\nu}(x')
    = \frac{\partial x^\rho}{\partial x'^\mu}
      \frac{\partial x^\sigma}{\partial x'^\nu}
      g_{\rho\sigma}(x).
  \end{equation}
\end{theorem}

\begin{proof}
$\partial_\mu\Theta$ transforms as $\partial'_\mu\Theta = (\partial x^\rho/\partial x'^\mu)\partial_\rho\Theta$.
The trace in~\eqref{eq:Gmunu} contracts the internal $\mathbb{C}\otimes\mathbb{H}$ indices only,
leaving two free covariant spacetime indices.  Applying the
chain rule twice gives the stated transformation law. $\square$
\end{proof}

\begin{tcolorbox}[colback=green!4!white,colframe=green!50!black,title=Consistency check: covariance]
\textbf{Verdict: VERIFIED.}
The construction \eqref{eq:Gmunu} is manifestly covariant; the real projection
preserves tensorial character.  No additional conditions are required.
\end{tcolorbox}

\section{Connection and Christoffel Symbols}\label{sec:connection}
% -----------------------------------------------------------------

Given the derived metric $g_{\mu\nu}$, the Levi-Civita connection is:
\begin{equation}\label{eq:christoffel}
  \Gamma^\lambda{}_{\mu\nu}
  \;=\;
  \tfrac{1}{2}\,g^{\lambda\rho}
  \!\left(\partial_\mu g_{\nu\rho} + \partial_\nu g_{\mu\rho} - \partial_\rho g_{\mu\nu}\right).
\end{equation}
This is the unique torsion-free, metric-compatible connection.

\begin{proposition}[Connection from biquaternionic metric]
The classical Christoffel symbols~\eqref{eq:christoffel} are the real projection
of the biquaternionic connection:
\begin{equation}
  \Gamma^\lambda{}_{\mu\nu} = \mathrm{Re}\!\left(\Omega^\lambda{}_{\mu\nu}\right),
\end{equation}
where $\Omega^\lambda{}_{\mu\nu}$ is the biquaternionic Levi-Civita connection
defined in \texttt{canonical/geometry/biquaternion\_connection.tex}.
\end{proposition}

\noindent\textbf{Source}: \texttt{canonical/geometry/connection.tex \S2},
\texttt{canonical/geometry/biquaternion\_connection.tex}.

\section{Curvature}\label{sec:curvature}
% -----------------------------------------------------------------

The Riemann curvature tensor is:
\begin{equation}\label{eq:riemann}
  R^\rho{}_{\sigma\mu\nu}
  \;=\;
  \partial_\mu\Gamma^\rho{}_{\nu\sigma}
  - \partial_\nu\Gamma^\rho{}_{\mu\sigma}
  + \Gamma^\rho{}_{\mu\lambda}\Gamma^\lambda{}_{\nu\sigma}
  - \Gamma^\rho{}_{\nu\lambda}\Gamma^\lambda{}_{\mu\sigma}.
\end{equation}

\noindent The Ricci tensor and Ricci scalar follow by contraction:
\begin{align}
  R_{\mu\nu} &= R^\rho{}_{\mu\rho\nu}, \\
  R &= g^{\mu\nu}R_{\mu\nu}.
\end{align}

\begin{proposition}[Curvature from $\Theta$]
The full Riemann curvature is a functional of $\Theta$:
\begin{equation}
  R^\rho{}_{\sigma\mu\nu}[\Theta]
  \;=\;
  \mathrm{Re}\!\left(\mathcal{R}^\rho{}_{\sigma\mu\nu}[\Theta]\right),
\end{equation}
where $\mathcal{R}^\rho{}_{\sigma\mu\nu}$ is the biquaternionic Riemann
curvature defined in \texttt{canonical/geometry/biquaternion\_curvature.tex}.
\end{proposition}

\noindent\textbf{Source}: \texttt{canonical/geometry/curvature.tex},
\texttt{canonical/geometry/biquaternion\_curvature.tex}.
