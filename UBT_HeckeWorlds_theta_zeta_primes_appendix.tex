\section*{Scope}
This appendix states the prime-sector (\emph{Hecke-Worlds}) axiom in the UBT framework, makes explicit the bridge
\emph{theta} $\leftrightarrow$ \emph{zeta} $\leftrightarrow$ \emph{primes}, and records the per-sector formula
\begin{equation}
\alpha_i^{-1} \;=\; p_i \;+\; \Delta_{\mathrm{CT},i}\,,
\end{equation}
while keeping all speculative interpretations (e.g.\ inter-sector transitions) out of scope.

\paragraph{Notation for $\Delta_{\mathrm{CT}}$.}
We denote by $\Delta_{\mathrm{CT}}$ the finite archimedean correction from the complex-time sector,
computed by matching the UBT current correlator to the QED vacuum polarization in the Thomson limit (two-loop), with no fitted parameters.

\section{Hecke-Worlds Axiom (Prime-Sector Selection)}
\begin{axiom}[Hecke-Worlds]\label{ax:hecke-worlds}
The physical sector of the UBT vacuum is selected by a Hecke (or level) sector indexed by a prime $p$.
In a fixed sector $i$ with anchor $p_i$, the fine-structure constant is uniquely determined (no free fits) by
\begin{equation}\label{eq:alpha-per-sector}
\alpha_i^{-1} \;=\; p_i \;+\; \Delta_{\mathrm{CT},i}\,,
\end{equation}
where $\Delta_{\mathrm{CT},i}$ is the archimedean correction obtained from the Ward identity and the Thomson limit matching.
\end{axiom}

\begin{proposition}[Uniqueness per sector]\label{prop:uniqueness}
Under place-covariance normalization (cf.\ Lemma~\ref{lem:place-cov}), and gauge invariance (Ward identity),
the ratio of UBT invariants defining $\alpha$ admits no additional free normalization constants; hence the value \eqref{eq:alpha-per-sector}
is \emph{unique} in each fixed prime sector $p_i$.
\end{proposition}

\begin{lemma}[Place-covariance normalization]\label{lem:place-cov}
Let $\Theta(q,\tau)$ be the biquaternion field with curvature coupling $\Lambda\!-\!R_\Theta$, and define
$\displaystyle \alpha = \mathrm{Re}[\Theta^\dagger D_\mu\Theta]/\mathrm{Im}[\Theta^\dagger D_\tau\Theta]$ in the archimedean limit.
Place-covariance fixes the relative normalization of these invariants and admits the prime-level representation $\alpha^{-1}=p+\Delta_{\mathrm{CT}}$.
\end{lemma}

\section{The \texorpdfstring{$\theta \leftrightarrow \zeta \leftrightarrow$}{theta <-> zeta <->} Primes Bridge}
\paragraph{Theta side (modular/automorphic).}
The Jacobi theta transforms under the modular group via $S:\tau\mapsto -1/\tau$ and $T:\tau\mapsto \tau+1$;
arithmetically, Hecke operators $T_p$ (indexed by primes $p$) act on modular/automorphic forms and possess eigenforms with eigenvalues $\lambda_p$.

\paragraph{Zeta side (Euler product).}
The Riemann zeta function admits the Euler product
\(
\zeta(s)=\prod_{p}(1-p^{-s})^{-1},
\)
so its local building blocks are precisely the primes. The Mellin transform of suitable theta-combinations reproduces the functional equation of $\zeta$, encoding the theta--zeta correspondence.

\paragraph{Bridge statement.}
In our UBT usage, the Hecke (prime-indexed) decomposition selects a \emph{prime sector} $p$ (``Hecke-World''),
and the per-sector physics evaluates the geometric/time invariant ratio \(\alpha\) in that sector, producing \(\alpha^{-1}=p+\Delta_{\mathrm{CT}}\).
Thus, \emph{theta} (via Hecke), \emph{zeta} (via Euler products), and \emph{primes} collectively justify the prime anchor while leaving
$\Delta_{\mathrm{CT}}$ to the archimedean Ward--Thomson matching.
