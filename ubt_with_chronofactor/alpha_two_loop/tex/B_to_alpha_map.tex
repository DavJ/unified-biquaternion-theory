\section{From $B$ to $\alpha$ in the Thomson Limit}
\label{sec:B2alpha}

\begin{lemma}[Ward Identity in CT]\label{lem:ward_ct}
In the complex-time (CT) renormalization scheme, the vertex and wavefunction renormalizations
satisfy $Z_1=Z_2$ to all orders provided the BRST symmetry is preserved.
\end{lemma}

\begin{proof}
Standard Ward identities lift to CT once the time fibre is included in the BRST complex.
Gauge variation of the 1PI functional yields $\partial_\mu \Gamma^\mu = S^{-1}(p)-S^{-1}(p')$,
leading to $Z_1=Z_2$. The presence of the $\psi$-fibre does not alter the cohomology;
see A1 and A3.
\end{proof}

\begin{lemma}[Thomson Limit and Scheme Independence]\label{lem:thomson}
Taking $q^2\to 0$ eliminates scheme $(\mu)$ and gauge $(\xi)$ dependence from the renormalized
coupling defined by low-energy scattering. In CT this continues to hold as the fibre contributions
cancel between $I_{\mathrm s}$ and $I_{\mathrm t}$ by A2–A3.
\end{lemma}

\begin{theorem}[Fit-free Identification $B \leftrightarrow \alpha$]\label{thm:B2alpha}
Let $B$ denote the geometric normalization relating the spatial invariant $I_{\mathrm s}$ to
the electromagnetic vertex in the Thomson limit. Under Lemmas~\ref{lem:ward_ct}–\ref{lem:thomson},
the map is
\begin{equation}\label{eq:B2alpha}
\alpha \;=\; \frac{e^2}{4\pi \hbar c}\, \mathcal R_{\mathrm{UBT}}\,,\qquad \mathcal R_{\mathrm{UBT}}=1\,.
\end{equation}
Thus there is no free normalization; the emergent $\alpha$ is fit-free.
\end{theorem}
